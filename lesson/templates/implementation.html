Tic-tac-toe

board = [[" ", " ", " "],
        [" ", " ", " "],
        [" ", " ", " "]]
count = 1
    
while count < 10:
  if count % 2 == 0: 
    player = "X"
  elif count % 2 == 1:
    player = "O"
  
  print("Player " + player + " , it is your turn.")
    
  while (True):
    row = int(input(" Pick a row: "))
    column = int(input("Pick a column: ")) 
    if (row in [0, 1, 2]) and (column in [0, 1,2]) and (board[row][column] not in ["X", "O" ]): 
      break 
    print("Invalid choice. Please try again!")
  
  board[row][column] = player 
  
  print("---+---+---")
  for r in range(3):
    print(board[r][0], " |" , board[r][1], "|" ,board[r][2])
    print("---+---+---")

  count += 1
